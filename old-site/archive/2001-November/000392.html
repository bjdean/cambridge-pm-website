Do other people find this behaviour very counter-intuitive:

---cut here---
#!/usr/bin/perl -w

use strict;

sub get_home ($) {
    my( $id ) = shift;

    open( PASSWD, "< /etc/passwd" ) or die $!;

    while( <PASSWD> ) {
	my( $id2, $passwd, $uid, $gid, $gecos, $home, $shell ) = split /:/;

	if( $id eq $id2 ) {
	    close( PASSWD ) or die $!;
	    return $home;
	}

    }

    close( PASSWD ) or die $!;
    return "User `$id' not found";
}

my @users = ( 'root', 'bin' );

my @homes = map { get_home( $_ ) } @users;

print "@users";
---cut here---

(Yes, I know there are better ways of doing this :-) )
